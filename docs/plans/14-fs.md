## 1. Overview & Goals

Unified virtual filesystem per project. Agent bisa read, write, list, move, delete files via built-in tools. File disimpan di storage adapter (S3/RustFS) tapi direpresentasikan sebagai virtual path tree di DB.

**Goals:**
- Virtual filesystem dengan path simulation (`/folder/subfolder/file.txt`)
- Adapter pattern — pluggable storage backend (S3 dulu, extensible)
- Credentials system untuk S3 config (reuse Plan 4)
- File manager UI di project
- Agent tools: read, write, list, move, delete
- Guard: hanya file types tertentu, max 5MB per file
- Per project scope — semua agent share filesystem yang sama

**Non-goals Plan 14:**
- Per-agent private filesystem
- File permissions per user/agent
- Symlinks
- File versioning / history
- Streaming upload/download (defer)
- Binary files (images, video, audio)

## 2. Allowed File Types

```typescript
export const ALLOWED_EXTENSIONS = [
  // Text
  '.txt', '.md', '.mdx', '.rst',
  // Web
  '.html', '.css', '.js', '.jsx', '.ts', '.tsx',
  // Programming
  '.py', '.rs', '.go', '.java', '.c', '.cpp', '.h',
  '.rb', '.php', '.swift', '.kt', '.cs', '.sh',
  // Config / Data
  '.json', '.yaml', '.yml', '.toml', '.env', '.ini',
  '.xml', '.csv', '.sql',
] as const

export const MAX_FILE_SIZE_BYTES = 5 * 1024 * 1024  // 5MB

export function isAllowedFile(filename: string, sizeBytes: number): {
  allowed: boolean
  reason?: string
} {
  const ext = path.extname(filename).toLowerCase()
  if (!ALLOWED_EXTENSIONS.includes(ext as any)) {
    return { allowed: false, reason: `File type ${ext} is not allowed` }
  }
  if (sizeBytes > MAX_FILE_SIZE_BYTES) {
    return { allowed: false, reason: `File size exceeds 5MB limit` }
  }
  return { allowed: true }
}
```

## 3. Virtual Path System

Path disimpan sebagai string di DB. Folder adalah virtual — tidak ada entitas folder di DB, folder exist kalau ada file dengan prefix tersebut.

```
/readme.md                    → file di root
/src/index.ts                 → file di folder /src
/src/utils/helper.ts          → file di nested folder
/docs/api/reference.md        → deep nested
```

### Path rules

```typescript
// Normalisasi path
function normalizePath(input: string): string {
  // Selalu mulai dengan /
  // Tidak boleh ada .. (path traversal)
  // Lowercase extension
  // Trim whitespace
  let p = input.trim()
  if (!p.startsWith('/')) p = '/' + p
  p = p.replace(/\/+/g, '/')          // collapse double slashes
  p = p.replace(/\.\./g, '')           // remove path traversal
  return p
}

// Extract folder dari path
function getFolderPath(filePath: string): string {
  return path.dirname(filePath)  // '/src/utils/helper.ts' → '/src/utils'
}

// List "folders" dari daftar paths (virtual)
function extractFolders(paths: string[]): string[] {
  const folders = new Set<string>()
  for (const p of paths) {
    const parts = p.split('/').filter(Boolean)
    for (let i = 0; i < parts.length - 1; i++) {
      folders.add('/' + parts.slice(0, i + 1).join('/'))
    }
  }
  return Array.from(folders).sort()
}
```

## 4. Storage Adapter Interface

```typescript
// packages/types/src/filesystem.ts

export interface FilesystemAdapter {
  readonly id: string
  readonly displayName: string

  // Upload file ke storage
  upload(key: string, content: string | Buffer, mimeType: string): Promise<void>

  // Download file dari storage
  download(key: string): Promise<Buffer>

  // Delete file dari storage
  delete(key: string): Promise<void>

  // Check apakah file exist
  exists(key: string): Promise<boolean>

  // Generate storage key dari project + virtual path
  buildKey(projectId: string, virtualPath: string): string
}

// S3 adapter implementation
export class S3FilesystemAdapter implements FilesystemAdapter {
  readonly id = 's3'
  readonly displayName = 'S3 / RustFS'

  constructor(private config: {
    endpoint: string
    accessKeyId: string
    secretAccessKey: string
    bucket: string
    region?: string
  }) {}

  buildKey(projectId: string, virtualPath: string): string {
    // projects/{project_id}{virtual_path}
    // e.g. projects/abc123/src/index.ts
    return `projects/${projectId}${virtualPath}`
  }

  async upload(key: string, content: string | Buffer, mimeType: string): Promise<void> {
    const client = this.getClient()
    await client.send(new PutObjectCommand({
      Bucket: this.config.bucket,
      Key: key,
      Body: content,
      ContentType: mimeType,
    }))
  }

  async download(key: string): Promise<Buffer> {
    const client = this.getClient()
    const result = await client.send(new GetObjectCommand({
      Bucket: this.config.bucket,
      Key: key,
    }))
    return Buffer.from(await result.Body!.transformToByteArray())
  }

  async delete(key: string): Promise<void> {
    const client = this.getClient()
    await client.send(new DeleteObjectCommand({
      Bucket: this.config.bucket,
      Key: key,
    }))
  }

  async exists(key: string): Promise<boolean> {
    try {
      const client = this.getClient()
      await client.send(new HeadObjectCommand({
        Bucket: this.config.bucket,
        Key: key,
      }))
      return true
    } catch {
      return false
    }
  }

  private getClient(): S3Client {
    return new S3Client({
      endpoint: this.config.endpoint,
      region: this.config.region ?? 'auto',
      credentials: {
        accessKeyId: this.config.accessKeyId,
        secretAccessKey: this.config.secretAccessKey,
      },
      forcePathStyle: true,  // required untuk RustFS/MinIO
    })
  }
}
```

### 5.2 `project_files` — virtual filesystem entries

```sql
CREATE TABLE project_files (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id   uuid NOT NULL REFERENCES projects(id) ON DELETE CASCADE,

  -- Virtual path (unique per project)
  path         text NOT NULL,              -- '/src/index.ts'
  name         text NOT NULL,              -- 'index.ts'
  folder_path  text NOT NULL,              -- '/src'
  extension    text NOT NULL,              -- '.ts'

  -- Storage
  storage_key  text NOT NULL,             -- key di storage adapter
  size_bytes   int NOT NULL DEFAULT 0,
  mime_type    text NOT NULL DEFAULT 'text/plain',

  -- Content cache (untuk file kecil, optional)
  -- Bisa null kalau file terlalu besar atau prefer fetch dari storage
  content_cache text,

  -- Metadata
  created_by   uuid REFERENCES users(id),  -- null kalau dibuat agent
  updated_by   uuid REFERENCES users(id),  -- null kalau diupdate agent

  created_at   timestamptz NOT NULL DEFAULT now(),
  updated_at   timestamptz NOT NULL DEFAULT now(),

  UNIQUE (project_id, path)
);

CREATE INDEX idx_files_project      ON project_files(project_id);
CREATE INDEX idx_files_folder       ON project_files(project_id, folder_path);
CREATE INDEX idx_files_extension    ON project_files(project_id, extension);
CREATE INDEX idx_files_updated      ON project_files(project_id, updated_at DESC);
```

## 6. Filesystem Service

```typescript
// apps/studio/server/src/filesystem/service.ts

export class FilesystemService {
  constructor(
    private projectId: string,
    private adapter: FilesystemAdapter,
  ) {}

  // List files dan virtual folders di path tertentu
  async list(folderPath: string = '/'): Promise<FilesystemEntry[]> {
    const normalizedFolder = normalizePath(folderPath)

    // Query files yang langsung di folder ini (bukan nested)
    const files = await db.query.projectFiles.findMany({
      where: and(
        eq(projectFiles.project_id, this.projectId),
        eq(projectFiles.folder_path, normalizedFolder),
      ),
      orderBy: [asc(projectFiles.name)],
    })

    // Extract virtual subfolders
    const allFiles = await db.query.projectFiles.findMany({
      where: and(
        eq(projectFiles.project_id, this.projectId),
        like(projectFiles.path, `${normalizedFolder === '/' ? '' : normalizedFolder}/%`),
      ),
      columns: { path: true },
    })

    const subfolders = extractImmediateSubfolders(
      allFiles.map(f => f.path),
      normalizedFolder,
    )

    return [
      ...subfolders.map(f => ({ type: 'folder' as const, path: f, name: path.basename(f) })),
      ...files.map(f => ({ type: 'file' as const, ...f })),
    ]
  }

  // Read file content
  async read(filePath: string): Promise<string> {
    const normalized = normalizePath(filePath)
    const file = await this.getFile(normalized)
    if (!file) throw new NotFoundError(`File not found: ${filePath}`)

    // Pakai cache kalau ada
    if (file.content_cache) return file.content_cache

    // Fetch dari storage
    const buffer = await this.adapter.download(file.storage_key)
    return buffer.toString('utf-8')
  }

  // Write file (create atau update)
  async write(
    filePath: string,
    content: string,
    createdBy?: string,
  ): Promise<ProjectFile> {
    const normalized = normalizePath(filePath)
    const filename = path.basename(normalized)
    const folder = path.dirname(normalized)
    const ext = path.extname(filename).toLowerCase()
    const sizeBytes = Buffer.byteLength(content, 'utf-8')

    // Validate
    const check = isAllowedFile(filename, sizeBytes)
    if (!check.allowed) throw new ValidationError(check.reason!)

    const storageKey = this.adapter.buildKey(this.projectId, normalized)
    const mimeType = getMimeType(ext)

    // Upload ke storage
    await this.adapter.upload(storageKey, content, mimeType)

    // Upsert di DB
    const existing = await this.getFile(normalized)

    if (existing) {
      await db.update(projectFiles)
        .set({
          size_bytes: sizeBytes,
          content_cache: sizeBytes <= 50_000 ? content : null,  // cache kalau < 50KB
          updated_by: createdBy ?? null,
          updated_at: new Date(),
        })
        .where(eq(projectFiles.id, existing.id))
      return { ...existing, size_bytes: sizeBytes, content_cache: content }
    }

    const [created] = await db.insert(projectFiles).values({
      project_id: this.projectId,
      path: normalized,
      name: filename,
      folder_path: folder,
      extension: ext,
      storage_key: storageKey,
      size_bytes: sizeBytes,
      mime_type: mimeType,
      content_cache: sizeBytes <= 50_000 ? content : null,
      created_by: createdBy ?? null,
    }).returning()

    await this.updateStats()
    return created
  }

  // Move / rename file
  async move(fromPath: string, toPath: string): Promise<void> {
    const from = normalizePath(fromPath)
    const to = normalizePath(toPath)

    const file = await this.getFile(from)
    if (!file) throw new NotFoundError(`File not found: ${fromPath}`)

    const existing = await this.getFile(to)
    if (existing) throw new ConflictError(`File already exists at: ${toPath}`)

    // Copy di storage
    const newKey = this.adapter.buildKey(this.projectId, to)
    const content = await this.adapter.download(file.storage_key)
    await this.adapter.upload(newKey, content, file.mime_type)

    // Delete old dari storage
    await this.adapter.delete(file.storage_key)

    // Update DB
    const filename = path.basename(to)
    await db.update(projectFiles)
      .set({
        path: to,
        name: filename,
        folder_path: path.dirname(to),
        extension: path.extname(filename).toLowerCase(),
        storage_key: newKey,
        updated_at: new Date(),
      })
      .where(eq(projectFiles.id, file.id))
  }

  // Delete file
  async delete(filePath: string): Promise<void> {
    const normalized = normalizePath(filePath)
    const file = await this.getFile(normalized)
    if (!file) throw new NotFoundError(`File not found: ${filePath}`)

    await this.adapter.delete(file.storage_key)
    await db.delete(projectFiles).where(eq(projectFiles.id, file.id))
    await this.updateStats()
  }

  // Delete folder (semua files dengan prefix)
  async deleteFolder(folderPath: string): Promise<number> {
    const normalized = normalizePath(folderPath)
    const files = await db.query.projectFiles.findMany({
      where: and(
        eq(projectFiles.project_id, this.projectId),
        or(
          eq(projectFiles.folder_path, normalized),
          like(projectFiles.path, `${normalized}/%`),
        ),
      ),
    })

    await Promise.all(files.map(f => this.adapter.delete(f.storage_key)))
    await db.delete(projectFiles)
      .where(inArray(projectFiles.id, files.map(f => f.id)))

    await this.updateStats()
    return files.length
  }

  private async getFile(normalizedPath: string) {
    return db.query.projectFiles.findFirst({
      where: and(
        eq(projectFiles.project_id, this.projectId),
        eq(projectFiles.path, normalizedPath),
      ),
    })
  }

  private async updateStats() {
    const result = await db.select({
      total_files: count(),
      total_size_bytes: sum(projectFiles.size_bytes),
    }).from(projectFiles)
      .where(eq(projectFiles.project_id, this.projectId))

    await db.update(projectFilesystemConfig)
      .set({
        total_files: result[0].total_files,
        total_size_bytes: Number(result[0].total_size_bytes ?? 0),
        updated_at: new Date(),
      })
      .where(eq(projectFilesystemConfig.project_id, this.projectId))
  }
}

// Factory — build service dari project config
export async function getFilesystemService(projectId: string): Promise<FilesystemService | null> {
  const config = await getFilesystemConfig(projectId)
  if (!config?.enabled || !config.credential_id) return null

  const credential = await resolveCredential(config.credential_id)
  const adapter = buildS3Adapter(credential.fields)

  return new FilesystemService(projectId, adapter)
}
```

## 7. Agent Tools

Built-in filesystem tools — aktif kalau project filesystem enabled. Di-register di `RuntimeManager.wakeUp()`.

```typescript
// apps/studio/server/src/filesystem/tools.ts

fs_list: tool({
  description: "List files and folders at a given path in the project filesystem",
  parameters: z.object({
    path: z.string().default('/').describe("Folder path to list. Default: root '/'"),
  }),
  execute: async ({ path }) => {
    const fs = await getFilesystemService(projectId)
    if (!fs) return { error: 'Filesystem is not configured for this project' }
    const entries = await fs.list(path)
    return { entries, count: entries.length }
  }
})

fs_read: tool({
  description: "Read the content of a file from the project filesystem",
  parameters: z.object({
    path: z.string().describe("Full file path, e.g. '/src/index.ts'"),
  }),
  execute: async ({ path }) => {
    const fs = await getFilesystemService(projectId)
    if (!fs) return { error: 'Filesystem is not configured for this project' }
    const content = await fs.read(path)
    return { path, content }
  }
})

fs_write: tool({
  description: "Write content to a file in the project filesystem. Creates the file if it doesn't exist, overwrites if it does.",
  parameters: z.object({
    path: z.string().describe("Full file path, e.g. '/src/utils/helper.ts'"),
    content: z.string().describe("File content to write"),
  }),
  execute: async ({ path, content }) => {
    const fs = await getFilesystemService(projectId)
    if (!fs) return { error: 'Filesystem is not configured for this project' }
    const file = await fs.write(path, content)
    return {
      success: true,
      path: file.path,
      size_bytes: file.size_bytes,
    }
  }
})

fs_move: tool({
  description: "Move or rename a file in the project filesystem",
  parameters: z.object({
    from: z.string().describe("Current file path"),
    to: z.string().describe("New file path"),
  }),
  execute: async ({ from, to }) => {
    const fs = await getFilesystemService(projectId)
    if (!fs) return { error: 'Filesystem is not configured for this project' }
    await fs.move(from, to)
    return { success: true, from, to }
  }
})

fs_delete: tool({
  description: "Delete a file from the project filesystem",
  parameters: z.object({
    path: z.string().describe("File path to delete"),

    if (!fs) return { error: 'Filesystem is not configured for this project' }

    return { success: true, path }
```
