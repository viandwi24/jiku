## 1. Motivasi Revisi

### Problem di Plan Sebelumnya

**1. Policy/ACL terlalu rigid**

`resource_type` dan `subject_type` adalah hardcoded enum — `'agent' | 'tool'` dan `'role' | 'permission'`. Developer yang pakai `@jiku/core` tidak bisa extend untuk resource atau subject type mereka sendiri.

**2. `checkAccess()` tidak pluggable**

Logic matching hardcoded ke `roles[]` dan `permissions[]` saja. Tidak ada cara untuk developer inject custom matching logic.

**3. `updateRules()` dipanggil tiap request**

Plan sebelumnya menyarankan `updateRules()` per chat request — ini menyebabkan potential race condition karena rules adalah shared state di runtime.

**4. `policy_rules` langsung FK ke `agent_id`**

Tidak reusable — satu policy tidak bisa di-attach ke banyak agent.

**5. Runtime tidak dianggap sebagai "world"**

Runtime belum diperlakukan sebagai long-lived workspace per project — tidak ada konsep wake up otomatis saat project dibuat.

### Goals Revisi

- `@jiku/core` policy system harus **se-generic mungkin** — string-based, pluggable matcher, support conditions
- Runtime adalah **permanent world** per project — wake up saat project dibuat, hidup selama project ada
- `runtime.run()` adalah **satu-satunya gerbang input** — chat, task, webhook, telegram, cron, semua lewat sini
- Policy di DB adalah **reusable entity** — satu policy bisa di-attach ke banyak agent
- Rules hanya di-update runtime kalau admin **memang mengubah policy** — bukan tiap request

### 2.2 Revisi `@jiku/core` — `resolver/access.ts`

#### `defaultSubjectMatcher`

```typescript
// packages/core/src/resolver/access.ts

export const defaultSubjectMatcher: SubjectMatcher = (
  rule: PolicyRule,
  caller: CallerContext
): boolean => {
  // Wildcard subject — match semua caller
  if (rule.subject === '*') return true

  switch (rule.subject_type) {
    case 'role':
      return caller.roles.includes(rule.subject)

    case 'permission':
      return caller.permissions.includes(rule.subject)

    case 'user':
      return caller.user_id === rule.subject

    default:
      // Cek di attributes — subject_type adalah attribute key
      // contoh: subject_type: 'plan', subject: 'pro'
      // → check caller.attributes?.plan === 'pro'
      const attr = caller.attributes?.[rule.subject_type]
      if (attr === undefined) return false
      return Array.isArray(attr)
        ? attr.includes(rule.subject)
        : attr === rule.subject
  }
}
```

#### `evaluateConditions`

```typescript
export function evaluateConditions(
  conditions: PolicyCondition[],
  caller: CallerContext
): boolean {
  // Semua conditions harus true (AND logic)
  return conditions.every(condition => {
    const value = resolveAttribute(condition.attribute, caller)
    if (value === undefined) return false

    switch (condition.operator) {
      case 'eq':
        return value === condition.value
      case 'not_eq':
        return value !== condition.value
      case 'in':
        return Array.isArray(condition.value)
          ? condition.value.includes(value as string)
          : false
      case 'not_in':
        return Array.isArray(condition.value)
          ? !condition.value.includes(value as string)
          : true
      case 'contains':
        return Array.isArray(value)
          ? value.includes(condition.value as string)
          : false
      case 'not_contains':
        return Array.isArray(value)
          ? !value.includes(condition.value as string)
          : true
    }
  })
}

// Resolve dot-notation path ke CallerContext
function resolveAttribute(
  path: string,
  caller: CallerContext
): string | string[] | undefined {
  // 'roles' → caller.roles
  // 'permissions' → caller.permissions
  // 'user_id' → caller.user_id
  // 'attributes.plan' → caller.attributes?.plan
  // 'user_data.company_id' → caller.user_data?.company_id

  const parts = path.split('.')

  let current: unknown = caller
  for (const part of parts) {
    if (current === null || current === undefined) return undefined
    current = (current as Record<string, unknown>)[part]
  }

  if (typeof current === 'string') return current
  if (Array.isArray(current)) return current as string[]
  return undefined
}
```

#### `checkAccess` — Revisi

```typescript
export function checkAccess(params: {
  resource_type: string
  resource_id: string
  caller: CallerContext
  rules: PolicyRule[]
  subject_matcher?: SubjectMatcher
}): boolean {
  const {
    resource_type,
    resource_id,
    caller,
    rules,
    subject_matcher = defaultSubjectMatcher,
  } = params

  // Filter rules yang relevan
  // resource_id '*' artinya apply ke semua resource dalam type ini
  const relevant = rules
    .filter(r =>
      r.resource_type === resource_type &&
      (r.resource_id === resource_id || r.resource_id === '*')
    )
    .sort((a, b) => (b.priority ?? 0) - (a.priority ?? 0))

  // Tidak ada rules → default ALLOW
  if (relevant.length === 0) return true

  for (const rule of relevant) {
    // Cek subject match
    const subjectMatch = subject_matcher(rule, caller)
    if (!subjectMatch) continue

    // Cek conditions (kalau ada) — semua harus true
    if (rule.conditions && rule.conditions.length > 0) {
      const conditionsMatch = evaluateConditions(rule.conditions, caller)
      if (!conditionsMatch) continue
    }

    // Subject match + conditions match → apply effect
    if (rule.effect === 'deny') return false
    if (rule.effect === 'allow') return true
  }

  // Ada rules tapi tidak ada yang match caller → DENY
  // Resource sudah di-restrict, caller tidak masuk kriteria
  return false
}
```

#### `resolveScope` — Revisi

```typescript
// Pass subject_matcher dari runtime options ke checkAccess
export function resolveScope(params: {
  caller: CallerContext
  agent: AgentDefinition
  rules: PolicyRule[]
  all_tools: ResolvedTool[]
  mode: AgentMode
  subject_matcher?: SubjectMatcher  // ← dari runtime options
}): ResolvedScope {
  const { caller, agent, rules, all_tools, mode, subject_matcher } = params

  const accessParams = { caller, rules, subject_matcher }

  // Check agent accessible
  const agentResource = `${agent.meta.id}:${mode}`
  const agentAccessible = checkAccess({
    ...accessParams,
    resource_type: 'agent',
    resource_id: agentResource,
  })

  if (!agentAccessible) {
    return {
      accessible: false,
      denial_reason: `No access to agent '${agent.meta.id}' in mode '${mode}'`,
      allowed_modes: [],
      active_tools: [],
      system_prompt: '',
    }
  }

  // Filter allowed modes
  const allowed_modes = agent.allowed_modes.filter(m =>
    checkAccess({
      ...accessParams,
      resource_type: 'agent',
      resource_id: `${agent.meta.id}:${m}`,
    })
  )

  // Filter tools
  const active_tools = all_tools
    .filter(tool => tool.modes.includes(mode))
    .filter(tool => {
      // Tool permission '*' → langsung allow, skip checkAccess
      if (tool.resolved_permission === '*') return true

      // Cek explicit rules dulu
      const hasExplicitRule = rules.some(r =>
        r.resource_type === 'tool' && r.resource_id === tool.resolved_id
      )

      if (hasExplicitRule) {
        // Ada explicit rule → evaluasi rule
        return checkAccess({
          ...accessParams,
          resource_type: 'tool',
          resource_id: tool.resolved_id,
        })
      }

      // Tidak ada explicit rule → fallback ke tool.resolved_permission
      // Cek apakah caller punya permission tersebut
      return caller.permissions.includes(tool.resolved_permission)
    })

  return {
    accessible: true,
    allowed_modes,
    active_tools,
    system_prompt: '',
  }
}
```

### 2.3 Revisi `JikuRuntime`

```typescript
// packages/core/src/runtime.ts

export class JikuRuntime {
  private _agents: Map<string, AgentRunner> = new Map()
  private _rules: PolicyRule[]
  private _subject_matcher: SubjectMatcher

  constructor(private options: JikuRuntimeOptions) {
    this._rules = options.rules ?? []
    this._subject_matcher = options.subject_matcher ?? defaultSubjectMatcher
  }

  // Hanya dipanggil kalau admin memang mengubah policy
  // Bukan per-request
  updateRules(rules: PolicyRule[]): void {
    this._rules = rules
  }

  async run(params: JikuRunParams): Promise<JikuRunResult> {
    const runner = this._agents.get(params.agent_id)
    if (!runner) throw new Error(`Agent '${params.agent_id}' not found`)

    return runner.run({
      ...params,
      rules: this._rules,
      subject_matcher: this._subject_matcher,
    })
  }
}
```

### 3.1 Runtime = Permanent World per Project

### 3.2 `JikuRuntimeManager` — Revisi

```typescript
// apps/studio/server/src/runtime/manager.ts

export class JikuRuntimeManager {
  private _runtimes: Map<string, JikuRuntime> = new Map()

  // Dipanggil saat server start — boot semua project yang ada
  async bootAll(): Promise<void> {
    const allProjects = await db.query.projects.findMany()
    await Promise.all(allProjects.map(p => this.wakeUp(p.id)))
    console.log(`[jiku] ✓ ${allProjects.length} project runtime(s) ready`)
  }

  // Dipanggil saat project dibuat — otomatis wake up
  async wakeUp(projectId: string): Promise<void> {
    if (this._runtimes.has(projectId)) return

    // Load agents + policy rules untuk project ini
    const [agentsData, rulesData] = await Promise.all([
      db.query.agents.findMany({ where: eq(agents.project_id, projectId) }),
      loadProjectPolicyRules(projectId),
    ])

    const runtime = new JikuRuntime({
      plugins: globalPluginLoader,
      storage: new StudioStorageAdapter(projectId),
      rules: rulesData,
      // Studio pakai default subject_matcher
      // support: role, permission, user, + attributes
    })

    // Register semua agents
    for (const agent of agentsData) {
      runtime.addAgent({
        meta: { id: agent.id, name: agent.name },
        base_prompt: agent.base_prompt,
        allowed_modes: agent.allowed_modes as AgentMode[],
        provider_id: agent.provider_id,
        model_id: agent.model_id,
      })
    }

    await runtime.boot()
    this._runtimes.set(projectId, runtime)
    console.log(`[jiku] ✓ Runtime woke up for project ${projectId}`)
  }

  // Dipanggil saat project dihapus
  async sleep(projectId: string): Promise<void> {
    const runtime = this._runtimes.get(projectId)
    if (!runtime) return
    await runtime.stop()
    this._runtimes.delete(projectId)
    console.log(`[jiku] ✓ Runtime slept for project ${projectId}`)
  }

  // Dipanggil saat admin ubah policy — bukan per-request
  async syncRules(projectId: string): Promise<void> {
    const runtime = this._runtimes.get(projectId)
    if (!runtime) return
    const rules = await loadProjectPolicyRules(projectId)
    runtime.updateRules(rules)
    console.log(`[jiku] ✓ Rules synced for project ${projectId}`)
  }

  // Dipanggil saat admin tambah/edit/hapus agent
  async syncAgent(projectId: string, agentId: string): Promise<void> {
    const runtime = this._runtimes.get(projectId)
    if (!runtime) return

    const agent = await db.query.agents.findFirst({
      where: eq(agents.id, agentId)
    })

    if (!agent) {
      runtime.removeAgent(agentId)
      return
    }

    runtime.removeAgent(agentId)
    runtime.addAgent({
      meta: { id: agent.id, name: agent.name },
      base_prompt: agent.base_prompt,
      allowed_modes: agent.allowed_modes as AgentMode[],
      provider_id: agent.provider_id,
      model_id: agent.model_id,
    })
  }

  getRuntime(projectId: string): JikuRuntime | undefined {
    return this._runtimes.get(projectId)
  }

  async stopAll(): Promise<void> {
    await Promise.all(
      [...this._runtimes.values()].map(r => r.stop())
    )
    this._runtimes.clear()
  }
}
```

```typescript
// apps/studio/server/src/runtime/caller.ts

export async function resolveCaller(
  userId: string,
  companyId: string,
  agentId: string,
  channel: string = 'web'
): Promise<CallerContext> {

  const [member, selfPolicy, user] = await Promise.all([
    db.query.company_members.findFirst({
      where: and(
        eq(company_members.user_id, userId),
        eq(company_members.company_id, companyId),
      ),
      with: {
        role: {
          with: {
            role_permissions: { with: { permission: true } }
          }
        }
      }
    }),
    db.query.agent_user_policies.findFirst({
      where: and(
        eq(agent_user_policies.agent_id, agentId),
        eq(agent_user_policies.user_id, userId),
      )
    }),
    db.query.users.findFirst({ where: eq(users.id, userId) })
  ])

  if (!member) throw new Error('Not a member of this company')

  const actualPermissions = member.role.role_permissions.map(rp => rp.permission.key)

  // Self-restriction — hanya bisa KURANGI, tidak bisa TAMBAH
  const effectivePermissions = selfPolicy && selfPolicy.allowed_permissions.length > 0
    ? actualPermissions.filter(p => selfPolicy.allowed_permissions.includes(p))
    : actualPermissions

  return {
    user_id: userId,
    roles: [member.role.name],
    permissions: effectivePermissions,
    user_data: {
      name: user!.name,
      email: user!.email,
      actual_permissions: actualPermissions,  // untuk UI self-restriction
    },
    // attributes — untuk policy conditions
    attributes: {
      channel,
      company_id: companyId,
      plan: 'free',   // future: load dari company subscription
    }
  }
}
```

```typescript
// apps/studio/server/src/runtime/caller.ts

export async function loadProjectPolicyRules(
  projectId: string
): Promise<PolicyRule[]> {
  // Load semua agents di project + policies mereka
  const agentsWithPolicies = await db.query.agents.findMany({
    where: eq(agents.project_id, projectId),
    with: {
      agent_policies: {
        orderBy: desc(agent_policies.priority),
        with: {
          policy: {
            with: { rules: true }
          }
        }
      }
    }
  })

  // Flatten + transform ke PolicyRule[]
  const allRules: PolicyRule[] = []

  for (const agent of agentsWithPolicies) {
    for (const ap of agent.agent_policies) {
      for (const rule of ap.policy.rules) {
        allRules.push({
          resource_type: rule.resource_type,
          resource_id: rule.resource_id === '*'
            ? '*'
            : rule.resource_id.startsWith(`${agent.id}:`)
              ? rule.resource_id
              : `${agent.id}:${rule.resource_id}`,
          subject_type: rule.subject_type,
          subject: rule.subject,
          effect: rule.effect as 'allow' | 'deny',
          priority: rule.priority ?? 0,
          conditions: (rule.conditions as PolicyCondition[]) ?? [],
        })
      }
    }
  }

  return allRules
}
```

```typescript
// apps/studio/server/src/ws/chat.ts

export function handleChatWS(
  ws: WSContext,
  conversationId: string,
  userId: string
) {
  ws.on('message', async (raw) => {
    const { input, agent_id, project_id, company_id } = JSON.parse(raw)

    // Resolve caller — per request, isolated
    const caller = await resolveCaller(userId, company_id, agent_id, 'web')

    // Get runtime — long-lived, sudah ada
    const runtime = runtimeManager.getRuntime(project_id)
    if (!runtime) {
      ws.send(JSON.stringify({ type: 'error', message: 'Runtime not ready' }))
      return
    }

    // Run — rules sudah ada di runtime, tidak perlu load ulang
    const result = await runtime.run({
      agent_id,
      caller,       // ← yang berubah per request hanya ini
      mode: 'chat',
      input,
      conversation_id: conversationId,
    })

    // Stream ke client
    for await (const chunk of result.stream) {
      ws.send(JSON.stringify(chunk))
    }

    ws.send(JSON.stringify({ type: 'done' }))
  })
}
```
